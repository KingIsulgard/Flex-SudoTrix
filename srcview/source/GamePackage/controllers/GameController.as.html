package GamePackage.controllers {
    import GamePackage.model.*;
    import GamePackage.screenitems.*;
    
    import flash.display.*;
    import flash.events.*;
    import flash.text.*;
    import flash.utils.*;
    
    /**
     * @author Gilles Lesire - 2009
     * This class represents the game itself: 
     * - builds up the screen for the game
     * - catch key events to move the active brick
     * - be able to pause/unpause the game
     * - places the brick in the correct arrays
     * - calculates the scores
     * - checks for full/faulty lines
     */    
    public class GameController extends Sprite {    
        // Events
        public static const GAME_FAILED:String = "GameFailed";
        public static const GAME_FINISHED:String = "GameFinished";
        public static const NEXT_LEVEL:String = "NextLevel";
        
        // Sound events
        public static const SOUND_EXPLOSION:String = "SoundExplosion";
        public static const SOUND_BLOCKED:String = "SoundBlocked";
        public static const SOUND_LAND:String = "SoundLand";
        public static const SOUND_FULL:String = "SoundFull";
        public static const SOUND_ERROR:String = "SoundError";
        public static const SOUND_DIE:String = "SoundDie";
                
        // Game elements
        private var playfield:Playfield;     
        private var scoreBoard:Scorebord;
        private var activeBrick:Brick;        
        private var popup:Popup;
        
        // Keyboard
        private var leftDown:Boolean = false;
        private var rightDown:Boolean = false;
        private var downDown:Boolean = false;
        private var upDown:Boolean = false;
        private var pDown:Boolean = false;
        private var yDown:Boolean = false;
        private var nDown:Boolean = false;
        private var spaceDown:Boolean = false;    
        
        // Score settings
        private const addBrickScore:Number = 10;
        private const removeErrorScore:Number = -10;
        private const addLineScore:Number = 100;
                
        // Game settings
        private var levelData:LevelData;
        private var maxLevel:int;
        
        // Game states            
        private var isPaused:Boolean = false;
        private var isCompleted:Boolean = false;
        private var isFailed:Boolean = false;
        private var isQuitting:Boolean = false;
        
        // Brick bounds
        private const boundLeft:Number = 56;
        private const boundRight:Number = 360; // width game - width 1 brick
        private const boundBottom:Number = 471;    // height game - height 1 brick
        private const startX:Number = 208;
        private const startY:Number = 54;
        
        // Brick lines and cubes in arrays
        private var bricks:Array = new Array();
        private var xLines:Array = new Array();
        private var yLines:Array = new Array();
        private var cubeLines:Array = new Array();   
        
        private var btnMain:Button;  
        
        // Available numbers
        private var availableNumbers:NumberController;
        
        public function GameController(levelData:LevelData, maxLevel:int) {
            this.levelData = levelData;
            this.maxLevel = maxLevel;
            
            playfield = new Playfield();
            addChild(playfield);
            
            scoreBoard = new Scorebord();
            updateScoreBoard();        
            addChild(scoreBoard);
                
            btnMain = new Button("Main menu");
            btnMain.addMouseEffects();
            btnMain.x = 525;
            btnMain.y = 505;
            btnMain.addEventListener(MouseEvent.CLICK, quitGame);

            addChild(btnMain);
            
            // Array vertical lines
            xLines["56"] = new Array();
            xLines["94"] = new Array();
            xLines["132"] = new Array();
            xLines["170"] = new Array();
            xLines["208"] = new Array();
            xLines["246"] = new Array();
            xLines["284"] = new Array();
            xLines["322"] = new Array();
            xLines["360"] = new Array();
            
            // Array horizontal lines
            yLines["167"] = new Array();
            yLines["205"] = new Array();
            yLines["243"] = new Array();
            yLines["281"] = new Array();
            yLines["319"] = new Array();
            yLines["357"] = new Array();
            yLines["395"] = new Array();
            yLines["433"] = new Array();
            yLines["471"] = new Array();
            
            // Array cubes
            cubeLines[0] = new Array();
            cubeLines[1] = new Array();
            cubeLines[2] = new Array();
            cubeLines[3] = new Array();
            cubeLines[4] = new Array();
            cubeLines[5] = new Array();
            cubeLines[6] = new Array();
            cubeLines[7] = new Array();
            cubeLines[8] = new Array();
            
            leftDown = false;
            rightDown = false;
            
            availableNumbers = new NumberController();            
            
            startGame();
        }    
        
        private function pauseGame():void {
            popup = new Popup("The game has been paused\nPress p to continue");
            addChild(popup);
            
            this.removeEventListener(Event.ENTER_FRAME, runGame);
            
            isPaused = true;
        }
        
        private function unPauseGame():void {
            removeChild(popup);
            
            this.addEventListener(Event.ENTER_FRAME, runGame);
            
            isPaused = false;                        
        }
        
        private function quitGame(event:MouseEvent):void {
            popup = new Popup("Are you sure you want to quit this game?\nY/N");            
            addChild(popup);

            this.removeEventListener(Event.ENTER_FRAME, runGame);
            
            isQuitting = true;
        }
        
        private function updateScoreBoard():void {
            scoreBoard.update(levelData);
        }
        
        private function updateErrors():void {
            levelData.currentErrors = 0;
            
            for each(var tempBrick:Brick in bricks) {
                if(tempBrick.getIsError())
                    levelData.currentErrors++;
            }
        }
        
        private function startGame():void {
            createBrick();
                
            this.addEventListener(Event.ENTER_FRAME, runGame);    
        }
        
        private function createBrick():void {
            activeBrick = new Brick(availableNumbers.getNumber());
            activeBrick.x = startX;
            activeBrick.y = startY;
            
            activeBrick.addEventListener(Event.ENTER_FRAME, moveBrick);
            addChild(activeBrick);
            
            bricks.push(activeBrick);
        }
        
        private function moveBrick(event:Event):void {
            if(leftDown) {
                if(activeBrick.x > boundLeft && !willCollide(activeBrick.x - activeBrick.width, activeBrick.y + activeBrick.height, activeBrick))
                    activeBrick.move("LEFT", activeBrick.width);
                else
                    dispatchEvent(new Event(SOUND_BLOCKED));
                setLeftDown(false);    // Deactivate to avoid unwanted repetition
            }
            if(rightDown) {
                if(activeBrick.x < boundRight && !willCollide(activeBrick.x + activeBrick.width, activeBrick.y + activeBrick.height, activeBrick))
                    activeBrick.move("RIGHT", activeBrick.width);
                else
                    dispatchEvent(new Event(SOUND_BLOCKED));
                setRightDown(false);    // Deactivate to avoid unwanted repetition
            }
            if(downDown) {
                levelData.currentSpeed = 10;
            } else {
                levelData.currentSpeed = levelData.dropSpeed;
            }
            if(upDown) {
                if(levelData.currentBombs > 0) {
                    if(activeBrick.getIsBomb())
                        activeBrick.unBomb();
                    else
                        activeBrick.bomb();
                }
                    
                setUpDown(false);    // Deactivate to avoid unwanted repetition
            }
            if(pDown) {
                if(isPaused) {
                    unPauseGame();                
                } else {
                    pauseGame();                        
                }
                            
                pDown = false; // Deactivate to avoid unwanted repetition
            }
            if(spaceDown) {
                if(isFailed) {
                    dispatchEvent(new Event(GAME_FAILED));
                }
                if(isCompleted) {
                    dispatchEvent(new Event(NEXT_LEVEL));    
                }

                spaceDown = false;
            } else if(yDown) {
                if(isQuitting) {
                    removeChild(popup);
            
                    dispatchEvent(new Event(GAME_FAILED));    
                }
                
                yDown = false;
                
            } else if(nDown) {
                if(isQuitting) {
                    removeChild(popup);
            
                    this.addEventListener(Event.ENTER_FRAME, runGame);
                }
                
                isQuitting = false;
                nDown = false;
            }
            
        }
        
        public function setLeftDown(setter:Boolean):void {
            leftDown = setter;
        }
        
        public function setRightDown(setter:Boolean):void {
            rightDown = setter;
        }
        
        public function setDownDown(setter:Boolean):void {
            downDown = setter;
        }
        
        public function setUpDown(setter:Boolean):void {
            upDown = setter;
        }    
        
        public function setPDown(setter:Boolean):void {
            pDown = setter;
        }
        
        public function setYDown(setter:Boolean):void {
            yDown = setter;
        }
        
        public function setNDown(setter:Boolean):void {
            nDown = setter;
        }
        
        public function setSpaceDown(setter:Boolean):void {
            spaceDown = setter;
        }
        
        private function runGame(e:Event):void {
            // If brick has landed
            if(activeBrick.y == boundBottom) {
                stopBrick(activeBrick);
                createBrick();
            
            // Adapt speed if necessairy
            } else if(boundBottom - activeBrick.y < levelData.currentSpeed) {
                activeBrick.move("DOWN", boundBottom - activeBrick.y);
            } else if(hitLowerBrick(activeBrick)) {    // Check for hitting lower brick
                stopBrick(activeBrick);
                createBrick();
            } else if(willCollide(activeBrick.x, activeBrick.y + activeBrick.height + levelData.currentSpeed, activeBrick)) {    // Check of de brick niet door een voorganger gaat schieten
                activeBrick.move("DOWN", activeBrick.height - ((activeBrick.y - startY) % activeBrick.height) - 1);
            } else {     // dropSpeed normal
                activeBrick.move("DOWN", levelData.currentSpeed);
            }                    
        }    
        
        private function stopBrick(tempBrick:Brick):void {        
            tempBrick.removeEventListener(Event.ENTER_FRAME, moveBrick);
            
            var bombedFound:Boolean = false;
            var bombedBrickX:int;
            var bombedBrickY:int;
            var bombedBrickBox:int;
                    
            // If bomb, remove the brick and the brick below
            if(tempBrick.getIsBomb()) {    
                // Play sound
                dispatchEvent(new Event(SOUND_EXPLOSION));
                            
                // Get brick that has been bombed
                for each(var checkBrick:Brick in bricks) {
                    if(checkBrick.x == tempBrick.x && checkBrick.y == tempBrick.y + checkBrick.height) {
                        if(!checkBrick.getIsDead()) {    
                            bombedFound = true;
                            bombedBrickX = checkBrick.x;
                            bombedBrickY = checkBrick.y;
                            bombedBrickBox = getBrickCube(checkBrick);
                            
                            availableNumbers.addNumber(checkBrick.getValue());                    
                            
                            removeBrick(checkBrick);
                        }
                    }    
                }                
                
                // If brick has been bombed, Recheck all errors        
                if(bombedFound) {            
                    removeBrickError(xLines[bombedBrickX.toString()]);
                    removeBrickError(yLines[bombedBrickY.toString()]);
                    removeBrickError(cubeLines[bombedBrickBox]);
                    
                    for each(var tempXLineArray:Array in xLines)
                        checkBrickError(tempXLineArray);
                    for each(var tempYLineArray:Array in yLines)
                        checkBrickError(tempYLineArray);    
                    for each(var tempCLineArray:Array in cubeLines)
                        checkBrickError(tempCLineArray);                                
                }
                
                availableNumbers.addNumber(tempBrick.getValue());
                        
                var explode : ExplodeDisolve = new ExplodeDisolve(tempBrick);
                addChild(explode);  // it's a sprite so you have to add it to the stage

                removeBomb(tempBrick);
                                
                levelData.currentBombs--;
                
            } else if(tempBrick.y <= startY + tempBrick.height) {
                availableNumbers.addNumber(tempBrick.getValue());
                
                removeBrick(tempBrick);
                
            } else if(tempBrick.y < startY + (tempBrick.height * 2)) {
                availableNumbers.addNumber(tempBrick.getValue());
                
                tempBrick.die();
                dispatchEvent(new Event(SOUND_DIE));
                    
            } else {
                var noError:Boolean = true;            
                
                // Check if the brick is already in the lines
                if(existsInLine(xLines[tempBrick.x.toString()], tempBrick.getValue()) || existsInLine(yLines[tempBrick.y.toString()], tempBrick.getValue()) || existsInLine(cubeLines[getBrickCube(tempBrick)], tempBrick.getValue())) {
                    noError = false;
                    
                    levelData.currentScore += removeErrorScore;
                    
                    xLines[tempBrick.x.toString()].push(tempBrick);
                    yLines[tempBrick.y.toString()].push(tempBrick);
                    cubeLines[getBrickCube(tempBrick)].push(tempBrick);
                    
                    setBrickError(xLines[tempBrick.x.toString()], tempBrick);
                    setBrickError(yLines[tempBrick.y.toString()], tempBrick);
                    setBrickError(cubeLines[getBrickCube(tempBrick)], tempBrick);                
                } else {
                    xLines[tempBrick.x.toString()].push(tempBrick);
                    yLines[tempBrick.y.toString()].push(tempBrick);
                    cubeLines[getBrickCube(tempBrick)].push(tempBrick);
                }
                    
                var noFull:Boolean = true;
                
                if(checkCompleteArray(xLines[tempBrick.x.toString()])) {
                    levelData.currentScore += addLineScore;
                    
                    setBrickFlash(xLines[tempBrick.x.toString()]);
                    
                    levelData.currentBombs++;
                    
                    noFull = false;
                }
                    
                if(checkCompleteArray(yLines[tempBrick.y.toString()])) {
                    levelData.currentScore += addLineScore;
                    
                    setBrickFlash(yLines[tempBrick.y.toString()]);
                    
                    levelData.currentBombs++;
                    
                    noFull = false;
                }
                
                if(checkCompleteArray(cubeLines[getBrickCube(tempBrick)])) {
                    levelData.currentScore += addLineScore;
                    
                    setBrickFlash(cubeLines[getBrickCube(tempBrick)]);
                    
                    levelData.currentBombs++;
                    
                    noFull = false;
                }
                
                // Play sound
                if(!noFull) {                
                    dispatchEvent(new Event(SOUND_FULL));    // Full line
                } else if(!noError) {
                    dispatchEvent(new Event(SOUND_ERROR));    // Error
                } else {
                    dispatchEvent(new Event(SOUND_LAND));    // Brick land
                }
                
                if(noFull && noError)
                    levelData.currentScore += addBrickScore;
            }
            
            // Gamescore can not go below 0
            if(levelData.currentScore < 0)
                levelData.currentScore = 0;
            
            // Check game
            updateErrors();
            updateScoreBoard();
            checkScoreLevelComplete();
            checkLevelFailed();            
        }

        private function checkLevelFailed():void {
            if(fieldIsFull()) {
                popup = new Popup("You failed to beat level " + levelData.level.toString() + "\nPress space to continue");
                
                addChild(popup);
                
                this.removeEventListener(Event.ENTER_FRAME, runGame);
                
                isFailed = true;        
            }
        }
        
        private function checkScoreLevelComplete():void {
            if(levelData.currentScore >= levelData.goal) {
                dispatchEvent(new Event(GAME_FINISHED));    
                
                if(levelData.level == maxLevel){
                    popup = new Popup("Congragulatons, you have finished\nall levels of SudoTrix!");
                } else {
                    popup = new Popup("You have finished level " + levelData.level.toString() + "\nPress space to continue");
                }        
                
                addChild(popup);
                
                this.removeEventListener(Event.ENTER_FRAME, runGame);
                
                isCompleted = true;                
            }
        }
        
        private function checkCompleteArray(tempArray:Array):Boolean {
            var isComplete:Boolean = true;
            
            for(var i:int = 1; i < 10; i++)
                if(!existsInLine(tempArray, i))
                    isComplete = false;
            
            return isComplete;
        }
        
        private function fieldIsFull():Boolean {
            var counter:int = 0;
            
            for each(var tempBrick:Brick in bricks) {
                if(!tempBrick.getIsDead()) {
                    counter++;
                }
            }
            
            if(counter == 81) {
                return true;
            } else {
                return false;
            }
        }
        
        private function getBrickCube(tempBrick:Brick):int {
            if(tempBrick.x < 170) { // First column
                if(tempBrick.y < 281) { // First row
                    return 0;    
                } else if(tempBrick.y < 395) { // Second row
                    return 1;
                } else { // Third row
                    return 2;
                }
            } else if(tempBrick.x < 284) { // Second column
                if(tempBrick.y < 281) { // First row
                    return 3;    
                } else if(tempBrick.y < 395) { // Second row
                    return 4;
                } else { // Third row
                    return 5;
                }
            } else { // Third column
                if(tempBrick.y < 281) { // First row
                    return 6;    
                } else if(tempBrick.y < 395) { // Second row
                    return 7;
                } else { // Third row
                    return 8;
                }
            }
        }
        
        private function removeBrickError(tempArray:Array):void {
            for each(var checkBrick:Brick in tempArray) {            
                checkBrick.error(false);
            }
        }
        
        private function checkBrickError(tempArray:Array):void {
            for each(var checkBrick:Brick in tempArray) {
                for each(var doubleBrick:Brick in tempArray) {
                    if(checkBrick != doubleBrick && checkBrick.getValue() == doubleBrick.getValue()) {
                        checkBrick.error(true);
                        doubleBrick.error(true);    
                    }
                }
            }
        }
        
        private function setBrickError(tempArray:Array, tempBrick:Brick):void {
            for each(var checkBrick:Brick in tempArray) {
                if(checkBrick.getValue() == tempBrick.getValue())  {
                    checkBrick.error(true);
                }
            }
        }
        
        private function setBrickFlash(tempArray:Array):void {
            for each(var tempBrick:Brick in tempArray) {
                tempBrick.flash();
            }
        }
                
        private function existsInLine(lineArray:Array, tempValue:int):Boolean {
            var existsIn:Boolean = false;
            
            for each(var checkBrick:Brick in lineArray) {
                if(checkBrick.getValue() == tempValue)
                    existsIn = true;
            }
            
            return existsIn;
        }
        
        private function removeBrick(tempBrick:Brick):void {
            bricks.splice(bricks.indexOf(tempBrick), 1);    
                
            if(tempBrick.y > startY + tempBrick.height * 2) {
                xLines[tempBrick.x.toString()].splice(xLines[tempBrick.x.toString()].indexOf(tempBrick), 1);
                yLines[tempBrick.y.toString()].splice(yLines[tempBrick.y.toString()].indexOf(tempBrick), 1);
                cubeLines[getBrickCube(tempBrick)].splice(cubeLines[getBrickCube(tempBrick)].indexOf(tempBrick), 1);
            }
                
            removeChild(tempBrick);
        }
        
        private function removeBomb(tempBrick:Brick):void {
            bricks.splice(bricks.indexOf(tempBrick), 1);    
                    
            removeChild(tempBrick);
        }
        
        private function hitLowerBrick(brick:Brick):Boolean {
            for each(var tempBrick:Brick in bricks) {
                if(brick.x == tempBrick.x && brick.y + brick.height == tempBrick.y) {
                    return true;
                }    
            }
            return false;
        }
        
        private function willCollide(tempX:int, tempY:int, self:Brick):Boolean {        
            for each(var tempBrick:Brick in bricks) {
                if(tempBrick != self) {
                    if(tempBrick.x == tempX && tempBrick.y <= tempY) {
                        return true;
                    }    
                }
            }
            return false;
        }
    }
}
