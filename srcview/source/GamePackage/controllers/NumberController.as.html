package GamePackage.controllers {
    /**
     * @author Gilles Lesire - 2009
     * This class contains all the numbers left to use: 
     * - generates the list of numbers 
     * - gives back a random number
     * - add numbers back when exploded
     * - shuffles the list when necassairy
     */    
    public class NumberController {
        private var numbers:Array = new Array();
        
        public function NumberController() {
            fill();
            shuffle();
        }
        
        private function fill():void {
            for(var i:int = 0; i < 9; i++) {
                for(var number:int = 1; number < 10; number++) {
                    numbers.push(number);
                }
            }
        }
        
        public function getNumber():int {
            return numbers.pop();
        }
        
        public function addNumber(number:int):void {
            numbers.push(number);
            shuffle();
        }    
        
        public function shuffle():void {
            var shuffledArray:Array = new Array();
            
            while(numbers.length > 0) {
                var randomNumber:int = Math.floor(Math.random() * numbers.length);
                shuffledArray.push(numbers[randomNumber]);
                numbers.splice(randomNumber, 1);
            }
            
            numbers = shuffledArray;
        }
    }
}
