package GamePackage.model
{
    import caurina.transitions.Tweener;
    
    import flash.display.*;
    import flash.filters.GlowFilter;
    import flash.utils.setTimeout;
    
    import mx.core.BitmapAsset;
    
    /**
     * @author Gilles Lesire - 2009
     * This class represents one block: 
     * - embeds graphic skin 
     * - set brick state
     * - move brick
     * - state can be asked
     * - can be asked for it's value
     */    
    public class Brick extends Sprite {
        [Embed(source="assets/skins/Bricks/Brick_1.png")] private var Brick_1:Class;
        [Embed(source="assets/skins/Bricks/Brick_2.png")] private var Brick_2:Class;
        [Embed(source="assets/skins/Bricks/Brick_3.png")] private var Brick_3:Class;
        [Embed(source="assets/skins/Bricks/Brick_4.png")] private var Brick_4:Class;
        [Embed(source="assets/skins/Bricks/Brick_5.png")] private var Brick_5:Class;
        [Embed(source="assets/skins/Bricks/Brick_6.png")] private var Brick_6:Class;
        [Embed(source="assets/skins/Bricks/Brick_7.png")] private var Brick_7:Class;
        [Embed(source="assets/skins/Bricks/Brick_8.png")] private var Brick_8:Class;
        [Embed(source="assets/skins/Bricks/Brick_9.png")] private var Brick_9:Class;
        [Embed(source="assets/skins/Bricks/Brick_Dead.png")] private var Brick_Dead:Class;
        [Embed(source="assets/skins/Bricks/Brick_Flash.png")] private var Brick_Flash:Class;
        [Embed(source="assets/skins/Bricks/Brick_Error.png")] private var Brick_Error:Class;
        [Embed(source="assets/skins/Bricks/Brick_Bomb.png")] private var Brick_Bomb:Class; 
        
        private var brick:BitmapAsset;  
        private var changedBrick:BitmapAsset;
        private var errorBrick:BitmapAsset;
        private var value:int = 0;
        
        private var isError:Boolean = false;
        private var isBomb:Boolean = false;
        private var isDead:Boolean = false;
        private var isFlashing:Boolean = false;
        
        public function Brick(brickType:int) {
            switch(brickType) {
                case     1:        brick = new Brick_1();             break;
                case     2:        brick = new Brick_2();             break;
                case     3:        brick = new Brick_3();             break;
                case     4:        brick = new Brick_4();             break;
                case     5:        brick = new Brick_5();             break;
                case     6:        brick = new Brick_6();             break;
                case     7:        brick = new Brick_7();             break;
                case     8:        brick = new Brick_8();             break;
                case     9:        brick = new Brick_9();             break;
                
                default:        brick = new Brick_Dead();        break;
            }
            
            value = brickType;
            
            brick.width = 38;
            brick.height = 38;

            addChild(brick);
        }
        
        public function error(setError:Boolean):void {
            if(setError) {
                if(!isError) {
                    isError = true;
                    
                    errorBrick = new Brick_Error();
                    errorBrick.width = 38;
                    errorBrick.height = 38;
                    
                    addChild(errorBrick);                    
                }            
            } else {
                if(isError) {
                    isError = false;
                    
                    removeChild(errorBrick);                    
                }
            }            
        }
        
        public function flash():void {
            if(!isFlashing) {
                setGlow(0xffffff);        
                
                changedBrick = new Brick_Flash();
                changedBrick.width = 38;
                changedBrick.height = 38;
                
                addChild(changedBrick);
                
                isFlashing = true;
                
                setTimeout(unFlash, 250);
            }
        }
        
        private function unFlash():void {
            isFlashing = false;
            
            if(changedBrick)
                removeChild(changedBrick);
                
            changedBrick = null;
            removeGlow();
        }
        
        private function setGlow(color:uint):void {
            var glow:GlowFilter = new GlowFilter(color);
            var glowFilter:Array = [glow];
            brick.filters = glowFilter;        
        }
        
        private function removeGlow():void {
            brick.filters = null;
        }
        
        public function die():void {            
            isDead = true;
            
            changedBrick = new Brick_Dead();
            changedBrick.alpha = 0;

            changedBrick.width = 38;
            changedBrick.height = 38;
            
            addChild(changedBrick);    
            Tweener.addTween(changedBrick, {alpha:100, time:300, transition:"easeOut", onComplete: removeOriginal});
        }
        
        private function removeOriginal():void {
            brick.visible = false;
        }

        public function bomb():void {
            brick.visible = false;
            isBomb = true;
            
            changedBrick = new Brick_Bomb();
            changedBrick.width = 38;
            changedBrick.height = 38;
            
            addChild(changedBrick);    
        }
        
        public function unBomb():void {
            if(isBomb) {
                brick.visible = true;
                changedBrick.visible = false;
                isBomb = false;    
            }
        }
        
        public function move(direction:String, distance:Number):void {
            if(direction == "DOWN") {
                this.y += distance;    
            } else if(direction == "LEFT") {
                this.x -= distance;
            } else if(direction == "RIGHT") {
                this.x += distance;
            }
        }        
        
        public function getIsBomb():Boolean {
            return isBomb;
        }
        
        public function getIsDead():Boolean {
            return isDead;
        }
        
        public function getValue():int {
            return value;
        }
        
        public function getIsError():Boolean {
            return isError;
        }
        
        public function getIsFlashing():Boolean {
            return isFlashing;
        }
    }
}
