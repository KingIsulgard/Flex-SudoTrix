package GamePackage.pages {
    import GamePackage.screenitems.*;
    
    import flash.display.*;
    import flash.events.*;
    import flash.text.*;
    import flash.utils.*;
    
    import mx.core.BitmapAsset;
    
    /**
     * @author Gilles Lesire - 2009
     * This class builds the how to play page
     * - creates the text and the logo
     * - set listener for the back to menu button
     */    
    public class Howto extends Sprite {
        [Embed(source="assets/skins/logo.png")] private var Logo:Class;
        [Embed(source="assets/fonts/showgothic.ttf", fontName="myFont", mimeType="application/x-font-truetype")]
        public static const FONT_MYFONT:String;
        
        public static const OPEN_MENU:String = "OpenMenu";
        
        private var formatText:TextFormat;
        private var text:TextField;
        private var btnMain:Button;  
        private var logo:BitmapAsset; 
        private var title:Button;
        
        public function Howto() {
            btnMain = new Button("Main menu");
            btnMain.addMouseEffects();
            btnMain.x = 525;
            btnMain.y = 505;
            btnMain.addEventListener(MouseEvent.CLICK, openMenu);
            
            logo = new Logo();        
            logo.x = 455;
            logo.y = 25;
            
            title = new Button("How to play");
            title.x = 54;
            title.y = 52;
            
            formatText = new TextFormat();
            formatText.align = TextFormatAlign.LEFT;
            formatText.color = 0x7BADDB;
            formatText.font = "myFont";
            formatText.size = 18;
            
            text = new TextField();        
            text.embedFonts = true;
            text.defaultTextFormat = formatText;
            text.wordWrap = true;
            text.text = "SudoTrix is a fun combinations of tetris and Sudoku.\n\n" + 
                    "Just as in Sudoku, you need to fill the grid with each " + 
                    "row, line and box with the number 1 to 9.\n\n" + 
                    "But it is not as easy as you think, you can't choose the numbers!\n\n" + 
                    "The goal of the game is to beat all levels.\n" + 
                    "Each level the score you have to get will rise.\n" + 
                    "\n" + 
                    "Use the arrow keys to navigate the brick.\n" + 
                    "You can turn a brick into a bomb\n" + 
                    "with the up key.\n\n" + 
                    "You have 5 bombs at the start of a new level, but you gain an extra one " + 
                    "for each full line you make.";
            text.x = 60;
            text.y = 120;
            text.width = 590;
            text.height = 385;
            
            addChild(btnMain);
            addChild(logo);
            addChild(title);
            addChild(text);
        }
        
        private function openMenu(event:Event):void {
            dispatchEvent(new Event(OPEN_MENU));    
        }
    }
}
