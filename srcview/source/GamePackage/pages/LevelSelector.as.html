package GamePackage.pages
{
    import GamePackage.events.*;
    import GamePackage.screenitems.*;
    
    import flash.display.*;
    import flash.events.*;
    import flash.text.*;
    
    import mx.core.BitmapAsset;
    
    /**
     * @author Gilles Lesire - 2009
     * This class builds the menu
     * - creates the menu items
     * - set listeners for menu items
     */     
    public class LevelSelector extends Sprite {
        [Embed(source="assets/skins/logo.png")] private var Logo:Class;
        [Embed(source="assets/skins/arrow_left.png")] private var ArrowLeft:Class;
        [Embed(source="assets/skins/arrow_right.png")] private var ArrowRight:Class;
        [Embed(source="assets/fonts/showgothic.ttf", fontName="myFont", mimeType="application/x-font-truetype")]
        public static const FONT_MYFONT:String;

        public static const OPEN_MENU:String = "OpenMenu";
            
        private var logo:BitmapAsset; 
        
        private var arrowLeft:ImageButton;
        private var arrowRight:ImageButton;
        
        private var btnMain:Button;  
        private var btnPlay:Button;
        private var titleText:FormattedText;
        private var levelText:FormattedText;
        
        private var selectedLevel:int;
        private var maxLevel:int;
        
        public function LevelSelector(selectedLevel:int, maxLevel:int) {
            this.selectedLevel = selectedLevel;
            this.maxLevel = maxLevel;
            
            logo = new Logo();            
            logo.x = 250;
            logo.y = 120;
            
            btnMain = new Button("Main menu");
            btnMain.addMouseEffects();
            btnMain.x = 525;
            btnMain.y = 505;
            btnMain.addEventListener(MouseEvent.CLICK, openMenu);
            
            btnPlay = new Button("Play");
            btnPlay.addMouseEffects();
            btnPlay.x = 525;
            btnPlay.y = 480;
            btnPlay.addEventListener(MouseEvent.CLICK, openGame);
            
            titleText = new FormattedText();
            titleText.color = 0x7BADDB;
            titleText.size = 20;
            titleText.align = TextFormatAlign.CENTER;
            titleText.text = "Select the level to play";
            titleText.x = 210;
            titleText.y = 240;
            titleText.width = 300;
            
            levelText = new FormattedText();        
            levelText.color = 0x7BADDB;
            levelText.size = 45;
            levelText.align = TextFormatAlign.CENTER;
            levelText.text = selectedLevel.toString();
            levelText.x = 305;
            levelText.y = 300;
            levelText.width = 100;
            
            arrowLeft = new ImageButton(new ArrowLeft(), new ArrowLeft());            
            arrowLeft.x = 265;
            arrowLeft.y = 295;
            arrowLeft.addEventListener(MouseEvent.CLICK, substractLevel);
            
            arrowRight = new ImageButton(new ArrowRight(), new ArrowRight());            
            arrowRight.x = 405;
            arrowRight.y = 295;
            arrowRight.addEventListener(MouseEvent.CLICK, addLevel);
            
            addChild(logo);
            addChild(btnMain);
            addChild(btnPlay);
            addChild(titleText);
            addChild(levelText);
            addChild(arrowLeft);
            addChild(arrowRight);
        }
        
        private function substractLevel(event:Event):void {
            if(selectedLevel > 1)
                selectedLevel--;
                
            levelText.text = selectedLevel.toString();
        }
        
        private function addLevel(event:Event):void {
            if(selectedLevel < maxLevel)
                selectedLevel++;
                
            levelText.text = selectedLevel.toString();
        }
        
        private function openMenu(event:Event):void {
            dispatchEvent(new Event(OPEN_MENU));    
        }
        
        private function openGame(event:Event):void {
            dispatchEvent(new GameEvent(GameEvent.START, selectedLevel));    
        }
    }
}
