package GamePackage.pages
{
    import GamePackage.screenitems.FormattedText;
    
    import flash.display.*;
    import flash.events.*;
    import flash.text.*;
    
    import mx.core.BitmapAsset;
    
    /**
     * @author Gilles Lesire - 2009
     * This class builds the menu
     * - creates the menu items
     * - set listeners for menu items
     */     
    public class Loading extends Sprite {
        [Embed(source="assets/skins/logo.png")] private var Logo:Class;
            
        private var logo:BitmapAsset; 
        private var text:FormattedText;
        
        public function Loading() {
            logo = new Logo();            
            logo.x = 250;
            logo.y = 120;
            
            text = new FormattedText();    
            text.text = "Loading.... \nPlease wait!";
            text.align = TextFormatAlign.CENTER;
            text.size = 18;            
            text.width = 200;
            text.color = 0x7BADDB;
            text.x = 250;
            text.y = 280;
            
            addChild(logo);
            addChild(text);
        }
    }
}
