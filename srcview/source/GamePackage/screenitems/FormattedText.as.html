package GamePackage.screenitems {
    import flash.display.*;
    import flash.text.*;
    
    /**
     * @author Gilles Lesire - 2009
     * This class builds a text
     * - creates the textfont
     * - set the text format
     * - creates a text
     */     
    public class FormattedText extends Sprite {
        [Embed(source="assets/fonts/showgothic.ttf", fontName="myFont", mimeType="application/x-font-truetype")]
        public static const FONT_MYFONT:String;
            
        private var formatText:TextFormat;
        private var textText:TextField;
        
        public function FormattedText() {
            formatText = new TextFormat();
            formatText.align = TextFormatAlign.CENTER;
            formatText.color = 0x000000;
            formatText.font = "myFont";
            
            textText = new TextField();
            textText.defaultTextFormat = formatText;
            textText.embedFonts = true;    
            
            addChild(textText);
        }
        
        public function get text():String {
            return textText.text;
        }
        
        public function set text(text:String):void {
            textText.text = text;
            textText.setTextFormat(formatText);
        }
        
        override public function get width():Number {
            return textText.width;
        }
        
        override public function set width(width:Number):void {
            textText.width = width;
        }
        
        public function set wordWrap(wordWrap:Boolean):void {
            textText.wordWrap = wordWrap;
        }
        
        public function set size(size:int):void {
            formatText.size = size;
            textText.setTextFormat(formatText);
        }
        
        public function set color(color:int):void {
            formatText.color = color;
            textText.setTextFormat(formatText);
        }
        
        public function set align(align:String):void {
            formatText.align = align;
            textText.setTextFormat(formatText);
        }
        
        public function set autoSize(autoSize:String):void {
            textText.autoSize = autoSize;
        }
        
        public function set selectable(selectable:Boolean):void {
            textText.selectable = selectable;
        }
        
        override public function set mouseEnabled(mouseEnabled:Boolean):void {
            textText.mouseEnabled = mouseEnabled;
        }
    }
}
