package GamePackage.screenitems
{    
    import flash.display.*;
    import flash.events.*;
    import flash.text.*;
    import flash.utils.*;
    
    import mx.core.BitmapAsset;
    
    /**
     * @author Gilles Lesire - 2009
     * This class builds a button
     * - creates the images
     * - change mouse pointer on hover
     * - creates mouseOver effect
     */     
    public class ImageButton extends Sprite {
        private var front:BitmapAsset;
        private var back:BitmapAsset;
        
        public function ImageButton(buttonFront:BitmapAsset, buttonBack:BitmapAsset) {
            front = buttonFront;
            back = buttonBack;
            
            addChild(buttonFront);
            
            addEventListener(MouseEvent.MOUSE_OVER, mouseOver);
            addEventListener(MouseEvent.MOUSE_OUT, mouseOut);
            
            buttonMode = true;
        }
        
        private function mouseOver(e:MouseEvent):void {
            addChild(back);
            removeChild(front);
        }
        
        private function mouseOut(e:MouseEvent):void {
            addChild(front);
            removeChild(back);
        }

    }
}
