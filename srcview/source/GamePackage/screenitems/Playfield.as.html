package GamePackage.screenitems {
    import flash.display.*;
    import flash.filters.GlowFilter;
    import flash.utils.setTimeout;
    
    import mx.core.BitmapAsset;
    
    /**
     * @author Gilles Lesire - 2009
     * This class builds the game interface
     * - creates a gamegrid
     * - adds logo on the right top
     * - adds button to main menu
     */    
    public class Playfield extends Sprite {
        [Embed(source="assets/skins/gamegrid.png")] private var GameGrid:Class;
        [Embed(source="assets/skins/logo.png")] private var Logo:Class;
        
        private var gamegrid:BitmapAsset; 
        private var logo:BitmapAsset; 
        
            
        public function Playfield() {
            gamegrid = new GameGrid();
            gamegrid.x = 54;
            gamegrid.y = 52;
            addChild(gamegrid);    
            
            logo = new Logo();
            logo.x = 455;
            logo.y = 25;
            addChild(logo);
        }

    }
}
