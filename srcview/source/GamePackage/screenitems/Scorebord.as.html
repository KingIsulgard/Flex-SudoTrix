package GamePackage.screenitems
{
    import GamePackage.model.*;
    
    import flash.display.*;
    import flash.text.*;
    import flash.utils.*;
    
    /**
     * @author Gilles Lesire - 2009
     * This class builds the game interface
     * - creates text fields and labels of game information
     * - updates text fields when called
     */    
    public class Scorebord extends Sprite {
        [Embed(source="assets/fonts/showgothic.ttf", fontName="myFont", mimeType="application/x-font-truetype")]
        public static const FONT_MYFONT:String;

        private var formatLabels:TextFormat;
        private var formatTexts:TextFormat;
        private var lblLevel:TextField;
        private var txtLevel:TextField;
        private var lblGoal:TextField;
        private var txtGoal:TextField;
        private var lblScore:TextField;
        private var txtScore:TextField;
        private var lblBombs:TextField;
        private var txtBombs:TextField;
        private var lblErrors:TextField;
        private var txtErrors:TextField;
        
        public function Scorebord() {
            formatLabels = new TextFormat();
            formatLabels.align = TextFormatAlign.LEFT;
            formatLabels.color = 0x7BADDB;
            formatLabels.font = "myFont";
            formatLabels.size = 18;
            
            formatTexts = new TextFormat();
            formatTexts.align = TextFormatAlign.RIGHT;
            formatTexts.color = 0x7BADDB;
            formatTexts.font = "myFont";
            formatTexts.size = 18;
            
            lblLevel = new TextField();
            txtLevel = new TextField();
            lblGoal = new TextField();
            txtGoal = new TextField();
            lblScore = new TextField();
            txtScore = new TextField();
            lblBombs = new TextField();        
            txtBombs = new TextField();        
            lblErrors = new TextField();
            txtErrors = new TextField();        
            
            lblLevel.embedFonts = true;
            lblLevel.defaultTextFormat = formatLabels;
            txtLevel.embedFonts = true;
            txtLevel.defaultTextFormat = formatTexts;
            lblGoal.embedFonts = true;
            lblGoal.defaultTextFormat = formatLabels;
            txtGoal.embedFonts = true;
            txtGoal.defaultTextFormat = formatTexts;
            lblScore.embedFonts = true;
            lblScore.defaultTextFormat = formatLabels;
            txtScore.embedFonts = true;
            txtScore.defaultTextFormat = formatTexts;
            lblBombs.embedFonts = true;
            lblBombs.defaultTextFormat = formatLabels;
            txtBombs.embedFonts = true;
            txtBombs.defaultTextFormat = formatTexts;
            lblErrors.embedFonts = true;
            lblErrors.defaultTextFormat = formatLabels;
            txtErrors.embedFonts = true;
            txtErrors.defaultTextFormat = formatTexts;            
            
            lblLevel.text = "Level";
            lblGoal.text = "Goal";
            lblScore.text = "Score";    
            lblBombs.text = "Bombs";
            lblErrors.text = "Errors";         
            
            lblLevel.x = 445;
            lblLevel.y = 135;
            
            txtLevel.x = 565;
            txtLevel.y = 135;
            
            lblGoal.x = 445;
            lblGoal.y = 165;
            
            txtGoal.x = 565;
            txtGoal.y = 165;
            
            lblScore.x = 445;
            lblScore.y = 225;
            
            txtScore.x = 565;
            txtScore.y = 225;
            
            lblBombs.x = 445;
            lblBombs.y = 255;
            
            txtBombs.x = 565;
            txtBombs.y = 255;
            
            lblErrors.x = 445;
            lblErrors.y = 285;
            
            txtErrors.x = 565;
            txtErrors.y = 285;
            
            addChild(lblLevel);
            addChild(txtLevel);
            addChild(lblGoal);
            addChild(txtGoal);
            addChild(lblScore);
            addChild(txtScore);
            addChild(lblBombs);
            addChild(txtBombs);
            addChild(lblErrors);
            addChild(txtErrors);
        }
        
        public function update(levelData:LevelData):void {
            txtLevel.text = levelData.level.toString();
            txtGoal.text = levelData.goal.toString();
            txtScore.text = levelData.currentScore.toString();
            txtBombs.text = levelData.currentBombs.toString();
            txtErrors.text = levelData.currentErrors.toString(); 
        }

    }
}
