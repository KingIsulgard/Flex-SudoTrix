package {
    import GamePackage.controllers.*;
    import GamePackage.events.*;
    import GamePackage.model.UserData;
    import GamePackage.pages.*;
    
    import flash.display.*;
    import flash.events.*;
    import flash.text.*;
    import flash.utils.*;
    
    import mx.core.BitmapAsset;
    
    /**
     * @author Gilles Lesire - 2009
     * This class is the main controller for the application: 
     * - creates and controls the underlying classes
     * - keeps data of what item is being shown
     * - creates event listeners to know when an object needs to be terminated
     * - propagate keyboard events to the underlying objects
     * - Catches thrown events and acts to it or propagates it
     */    
    public class SudoTrix extends Sprite {
        [Embed(source="assets/skins/background.png")] private var Background:Class;
        private var background:BitmapAsset;     
        
        // Create items to show
        private var game:GameController;
        private var menu:Menu;
        private var credits:Credits;
        private var howto:Howto;
        private var loading:Loading;
        private var levelSelector:LevelSelector;
        
        private var userData:UserData;
        
        private var windowOpened:String;
        
        private var soundController:SoundController;
        private var levelsController:LevelsController;    
        private var cookieController:CookieController;
        
        // Keyboard codes
        private static const LEFT_ARROW:int = 37;
        private static const UP_ARROW:int = 38;
        private static const RIGHT_ARROW:int = 39;
        private static const DOWN_ARROW:int = 40;
        private static const P_KEY:int = 80;
        private static const Y_KEY:int = 89;
        private static const N_KEY:int = 78;
        private static const SPACE_KEY:int = 32;
        
        public function SudoTrix() {
            stage.scaleMode = StageScaleMode.NO_SCALE;
            stage.align = StageAlign.TOP_LEFT;
            
            // Set background to game
            background = new Background();
            addChild(background);
            
            // Set keyboard listeners
            stage.addEventListener(KeyboardEvent.KEY_DOWN, checkKeysDown);    
            stage.addEventListener(KeyboardEvent.KEY_UP, checkKeysUp);
            
            // Load cookies
            cookieController = new CookieController()
            
            // Get userData
            userData = cookieController.read();        
            
            // Load sounds
            soundController = new SoundController();
            soundController.addEventListener(SoundController.SOUNDS_LOADED, onLoaded);
            
            // Get level information
            levelsController = new LevelsController();
            levelsController.addEventListener(LevelsController.LEVELS_LOADED, onLoaded);
            
            // Display loading message
            loading = new Loading();
            addChild(loading);
        } 
        
        private function onLoaded(event:Event):void {
            if(soundController.isLoaded && levelsController.isLoaded)
                allLoaded();
        }
        
        private function allLoaded():void {
            // Remove loading message
            removeChild(loading);
            
            // Open menu
            openMenu();    
        }
        
        private function removeAll():void {
            if(windowOpened == "menu")
                removeChild(menu);
            if(windowOpened == "game")
                removeChild(game);
            if(windowOpened == "credits")
                removeChild(credits);
            if(windowOpened == "howto")
                removeChild(howto);    
            if(windowOpened == "selector")
                removeChild(levelSelector);                
        }
        
        private function openMenu():void {
            removeAll();
            
            windowOpened = "menu";
            
            menu = new Menu();
            menu.addEventListener(Menu.OPEN_GAME, onGameOpen);
            menu.addEventListener(Menu.OPEN_CREDITS, onCreditsOpen);
            menu.addEventListener(Menu.OPEN_HOWTO, onHowtoOpen);
            addChild(menu);            
        }
        
        private function openCredits():void {
            removeAll();
            
            windowOpened = "credits";
            
            credits = new Credits();
            credits.addEventListener(Credits.OPEN_MENU, onOpenMenu);
            addChild(credits);
        }
        
        private function openHowto():void {
            removeAll();
            
            windowOpened = "howto";
            
            howto = new Howto();
            howto.addEventListener(Howto.OPEN_MENU, onOpenMenu);
            addChild(howto);
        }
        
        private function openLevelSelector():void {
            // Clear screen    
            removeAll();
            
            windowOpened = "selector";
            
            // Create a levelselector
            levelSelector = new LevelSelector(userData.gameLevel, userData.achievedLevel);
            
            // Add event listeners
            levelSelector.addEventListener(GameEvent.START, onStartGame);
            levelSelector.addEventListener(LevelSelector.OPEN_MENU, onOpenMenu);
            
            // Add to screen
            addChild(levelSelector);
        }
        
        private function onStartGame(event:GameEvent):void {
            // Set level
            userData.gameLevel = event.level;
            
            // Start game of the selected level
            startGame();
        }
        
        private function startGame():void {
            removeAll();
            
            if(userData.gameLevel > levelsController.getAmountLevels()) {
                openMenu();
            } else {            
                game = new GameController(levelsController.getLevel(userData.gameLevel), levelsController.getAmountLevels());
                addChild(game);
                windowOpened = "game";
                
                game.addEventListener(GameController.GAME_FAILED, onOpenMenu);
                game.addEventListener(GameController.GAME_FINISHED, onGameFinished);
                game.addEventListener(GameController.NEXT_LEVEL, onGameOpen);
                
                game.addEventListener(GameController.SOUND_EXPLOSION, onSoundEvent);
                game.addEventListener(GameController.SOUND_BLOCKED, onSoundEvent);
                game.addEventListener(GameController.SOUND_LAND, onSoundEvent);
                game.addEventListener(GameController.SOUND_FULL, onSoundEvent);
                game.addEventListener(GameController.SOUND_ERROR, onSoundEvent);
                game.addEventListener(GameController.SOUND_DIE, onSoundEvent);
                game.addEventListener(GameController.SOUND_ERROR, onSoundEvent);
            }
        }
        
        private function onOpenMenu(event:Event):void {
            openMenu();
        }
        
        private function onGameFinished(event:Event):void {
            if(userData.gameLevel == userData.achievedLevel) {
                if(userData.achievedLevel < levelsController.getAmountLevels())
                    userData.achievedLevel++;
                    
                userData.gameLevel = userData.achievedLevel;
                
                cookieController.save(userData);
            }
        }
        
        private function onGameOpen(event:Event):void {
            openLevelSelector();
        }
        
        private function onCreditsOpen(event:Event):void {
            openCredits();
        }
        
        private function onHowtoOpen(event:Event):void {
            openHowto();
        }
        
        private function onSoundEvent(event:Event):void {
            if(event.type == GameController.SOUND_EXPLOSION)
                soundController.playExplosion();
            if(event.type == GameController.SOUND_LAND)
                soundController.playLand();
            if(event.type == GameController.SOUND_BLOCKED)
                soundController.playBlocked();
            if(event.type == GameController.SOUND_FULL)
                soundController.playFull();
            if(event.type == GameController.SOUND_ERROR)
                soundController.playError();
            if(event.type == GameController.SOUND_DIE)
                soundController.playDie();
        }
        
        private function checkKeysDown(event:KeyboardEvent):void {        
            if(windowOpened == "game") {
                if(event.keyCode == LEFT_ARROW) {
                    game.setLeftDown(true);
                }
                if(event.keyCode == RIGHT_ARROW) {
                    game.setRightDown(true);
                }
                if(event.keyCode == DOWN_ARROW) {
                    game.setDownDown(true);
                }
                if(event.keyCode == UP_ARROW) {
                    game.setUpDown(true);
                }
                if(event.keyCode == P_KEY) {
                    game.setPDown(true);
                }
                if(event.keyCode == Y_KEY) {
                    game.setYDown(true);
                }
                if(event.keyCode == N_KEY) {
                    game.setNDown(true);
                }
                if(event.keyCode == SPACE_KEY) {
                    game.setSpaceDown(true);
                }
            }
            
        }
        
        private function checkKeysUp(event:KeyboardEvent):void {
            if(event.keyCode == DOWN_ARROW) {
                game.setDownDown(false);
            }
        }

    }
}
